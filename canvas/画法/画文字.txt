**************描绘文字

fillText(text,x,y,[maxWidth])--------------------位置，最大宽度----------填充文本

strokeText(text,x,y,[maxWidth])------------------------------------------描边文本

font=value--------------文本样式，默认10px sans-serif

textAlign=value---------文本对齐选项，start（默认）,end,left,right,center

textBaseline=value------基线对齐选项，top,hanging,middle,alphabetic（默认）,ideographic,bottom

direction=value---------文本方向，ltr,rtl,inherit（默认）

measureText('Hellow')--测量文本长度
